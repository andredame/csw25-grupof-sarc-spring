Table usuario {
  id int [pk]
  email varchar
  nome varchar
  data_nascimento date
  sexo varchar
  telefone varchar
  numero varchar
  id_perfil int [ref: > perfil.id]
}

Table perfil {
  id int [pk]
  nome varchar
}

Table disciplina {
  id int [pk]
  codigo varchar
  nome varchar
  creditos int
  semestre varchar
  programa text
}

Table turma {
  id int [pk]
  numero varchar
  disciplina_id int [ref: > disciplina.id]
  semestre varchar
  professor_id int [ref: > usuario.id]
  horario varchar
  vagas int
}


Table chamada {
  id int [pk]
  aula_id int [ref: > aula.id]
  aluno_id int [ref: > usuario.id]
  presente boolean
}

Table aula {
  id int [pk]
  turma_id int [ref: > turma.id]
  data date
  sala_id int [ref: > sala.id]
}

Table sala {
  id int [pk]
  nome varchar
  capacidade int
  andar varchar
  predio_id int [ref: > predio.id]
}

Table predio {
  id int [pk]
  nome varchar
  rua varchar
  numero varchar
  complemento varchar
  bairro varchar
  cidade varchar
  uf varchar
  cep varchar
}

Table reserva {
  id int [pk]
  aula_id int [ref: > aula.id]
  recurso_id int [ref: > recurso.id]
  observacao text
}

Table recurso {
  id int [pk]
  descricao text
  status varchar
  id_tipo_recurso int [ref: > tipo_recurso.id]
}

Table tipo_recurso {
  id int [pk]
  nome varchar
}

Table conteudo_aula {
  id int [pk]
  aula_id int [ref: > aula.id]
  conteudo text
}

Table turma_aluno {
  id int [pk]
  turma_id int [ref: > turma.id]
  aluno_id int [ref: > usuario.id]
}
